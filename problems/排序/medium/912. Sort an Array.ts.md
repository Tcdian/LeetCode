###

给你一个整数数组 `nums`，请你将该数组升序排列。

#### 示例 1：

```
输入：nums = [5,2,3,1]
输出：[1,2,3,5]
```

#### 示例 2：

```
输入：nums = [5,1,1,2,0,0]
输出：[0,0,1,1,2,5]
```

#### 提示：

-   `1 <= nums.length <= 50000`
-   `-50000 <= nums[i] <= 50000`

#

#### 方法一：归并排序

```ts
function sortArray(nums: number[]): number[] {
    const helper: number[] = new Array(nums.length);
    mergeSort(nums, 0, nums.length - 1);
    return nums;

    function mergeSort(nums: number[], left: number, right: number) {
        if (left >= right) {
            return;
        }
        const mid = (left + right) >> 1;
        mergeSort(nums, left, mid);
        mergeSort(nums, mid + 1, right);
        merge(nums, left, mid, right);
    }

    function merge(nums: number[], left: number, mid: number, right: number) {
        for (let i = left; i <= right; i++) {
            helper[i] = nums[i];
        }
        let i = left;
        let j = mid + 1;
        for (let k = left; k <= right; k++) {
            if (i > mid) {
                nums[k] = helper[j];
                j++;
            } else if (j > right) {
                nums[k] = helper[i];
                i++;
            } else if (helper[i] <= helper[j]) {
                nums[k] = helper[i];
                i++;
            } else {
                nums[k] = helper[j];
                j++;
            }
        }
    }
}
```

#### 方法二：快速排序

```ts
function sortArray(nums: number[]): number[] {
    quickSort(nums, 0, nums.length - 1);
    return nums;

    function quickSort(nums: number[], left: number, right: number) {
        if (left >= right) {
            return;
        }
        const pointIndex = ((Math.random() * (right - left + 1)) | 0) + left;
        const pointValue = nums[pointIndex];
        swap(nums, pointIndex, right);
        let separator = left - 1;
        for (let i = left; i < right; i++) {
            if (nums[i] < pointValue) {
                swap(nums, ++separator, i);
            }
        }
        swap(nums, ++separator, right);
        quickSort(nums, left, separator - 1);
        quickSort(nums, separator + 1, right);
    }

    function swap(nums: number[], i: number, j: number) {
        const temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }
}
```

#### 方法三：堆排序

```ts
function sortArray(nums: number[]): number[] {
    heapSort(nums);
    return nums;

    function heapSort(nums: number[]) {
        for (let i = (nums.length - 2) >> 1; i >= 0; i--) {
            siftDown(nums, i, nums.length);
        }
        for (let i = nums.length - 1; i >= 0; i--) {
            swap(nums, 0, i);
            siftDown(nums, 0, i);
        }
    }

    function siftDown(nums: number[], index: number, k: number) {
        while (2 * index + 1 < k) {
            let maxValueIndex = 2 * index + 1;
            const rightChildIndex = 2 * index + 2;
            if (rightChildIndex < k && nums[maxValueIndex] < nums[rightChildIndex]) {
                maxValueIndex = rightChildIndex;
            }
            if (nums[index] > nums[maxValueIndex]) {
                break;
            }
            swap(nums, index, maxValueIndex);
            index = maxValueIndex;
        }
    }

    function swap(nums: number[], i: number, j: number) {
        const temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }
}
```

#### 方法四：希尔排序

```ts
function sortArray(nums: number[]): number[] {
    shellSort(nums);
    return nums;

    function shellSort(nums: number[]) {
        let h = 1;
        // 步长序列：1，4，13，40，...
        while (h < nums.length) {
            h = h * 3 + 1;
        }
        while (h >= 1) {
            for (let i = h; i < nums.length; i++) {
                const current = nums[i];
                let j: number;
                for (j = i; j - h >= 0 && current < nums[j - h]; j -= h) {
                    nums[j] = nums[j - h];
                }
                nums[j] = current;
            }
            h = (h / 3) | 0;
        }
    }
}
```

#### 方法五：计数排序

```ts
function sortArray(nums: number[]): number[] {
    const min = Math.min(...nums);
    const max = Math.max(...nums);
    const R = max - min + 1;
    const counts: number[] = new Array(R).fill(0);
    for (let i = 0; i < nums.length; i++) {
        counts[nums[i] - min]++;
    }
    const indexes: number[] = new Array(R + 1).fill(0);
    for (let i = 0; i < R; i++) {
        indexes[i + 1] = indexes[i] + counts[i];
    }
    const sorted: number[] = new Array(nums.length);
    for (let i = 0; i < nums.length; i++) {
        sorted[indexes[nums[i] - min]] = nums[i];
        indexes[nums[i] - min]++;
    }
    for (let i = 0; i < nums.length; i++) {
        nums[i] = sorted[i];
    }
    return nums;
}
```

#### 方法六：基数排序

-   LSD（Least significant digital）

```ts
function sortArray(nums: number[]): number[] {
    LSDSort(nums, 0);
    return nums;

    function LSDSort(nums: number[], pow: number) {
        const helper: number[] = new Array(nums.length);
        while (pow < 5) {
            let R = 9 + 9 + 1; // -9 ~ 9
            const counts: number[] = new Array(R).fill(0);
            for (let i = 0; i < nums.length; i++) {
                const p = (((nums[i] / 10 ** pow) | 0) % 10) + 9;
                counts[p]++;
            }
            const indexes: number[] = new Array(R + 1).fill(0);
            for (let i = 0; i < R; i++) {
                indexes[i + 1] = indexes[i] + counts[i];
            }
            for (let i = 0; i < nums.length; i++) {
                const p = (((nums[i] / 10 ** pow) | 0) % 10) + 9;
                helper[indexes[p]] = nums[i];
                indexes[p]++;
            }
            for (let i = 0; i < nums.length; i++) {
                nums[i] = helper[i];
            }
            pow++;
        }
    }
}
```

-   MSD（Most significant digital）

```ts
function sortArray(nums: number[]): number[] {
    MSDSort(nums, 0, nums.length - 1, 4);
    return nums;

    function MSDSort(nums: number[], left: number, right: number, pow: number) {
        if (left >= right || pow < 0) {
            return;
        }
        let R = 9 + 9 + 1; // -9 ~ 9
        const counts: number[] = new Array(R).fill(0);
        for (let i = left; i <= right; i++) {
            const p = (((nums[i] / 10 ** pow) | 0) % 10) + 9;
            counts[p]++;
        }
        const indexes: number[] = new Array(R + 1).fill(0);
        for (let i = 0; i < R; i++) {
            indexes[i + 1] = indexes[i] + counts[i];
        }
        const helper: number[] = new Array(right - left + 1);
        for (let i = left; i <= right; i++) {
            const p = (((nums[i] / 10 ** pow) | 0) % 10) + 9;
            helper[indexes[p]] = nums[i];
            indexes[p]++;
        }
        for (let i = left; i <= right; i++) {
            nums[i] = helper[i - left];
        }
        let start = 0;
        for (let i = 0; i <= R; i++) {
            MSDSort(nums, start + left, indexes[i] - 1 + left, pow - 1);
            start = indexes[i];
        }
    }
}
```

#### 方法七：桶排序

```ts
function sortArray(nums: number[]): number[] {
    bucketSort(nums, 100);
    return nums;

    function bucketSort(nums: number[], c: number) {
        const min = Math.min(...nums);
        const max = Math.max(...nums);
        const buckets: number[][] = Array.from(new Array((((max - min) / c) | 0) + 1), () => []);
        for (let i = 0; i < nums.length; i++) {
            buckets[((nums[i] - min) / c) | 0].push(nums[i]);
        }
        for (let i = 0; i < buckets.length; i++) {
            buckets[i].sort((a, b) => a - b);
        }
        let index = 0;
        for (let i = 0; i < buckets.length; i++) {
            for (let j = 0; j < buckets[i].length; j++) {
                nums[index++] = buckets[i][j];
            }
        }
    }
}
```
